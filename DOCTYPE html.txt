<script>
const resourceGrid = document.getElementById('resourceGrid');

document.addEventListener('DOMContentLoaded', getResources);

// --- AJOUT ---
document.getElementById('addBtn').addEventListener('click', () => {
const title = document.getElementById('resTitle').value.trim();
const link = document.getElementById('resLink').value.trim();

if (!title || !link.startsWith('http')) return alert("Données invalides !");

const resource = { title, link, id: Date.now() };
createCard(resource);
saveLocalResource(resource);
document.getElementById('resTitle').value = '';
document.getElementById('resLink').value = '';
});

// --- RECHERCHE ---
document.getElementById('searchInput').addEventListener('input', (e) => {
const term = e.target.value.toLowerCase();
document.querySelectorAll('.card').forEach(card => {
const isMatch = card.querySelector('h3').innerText.toLowerCase().includes(term);
card.classList.toggle('hidden', !isMatch);
});
});

// --- CRÉATION DE CARTE ---
function createCard(resource) {
const card = document.createElement('div');
card.classList.add('card');
card.setAttribute('data-id', resource.id);

card.innerHTML = `
<h3>${resource.title}</h3>
<div class="card-actions">
<button class="btn-action copy-btn" onclick="copyLink('${resource.link}', this)">Copier</button>
<a href="${resource.link}" target="_blank" style="flex:1"><button class="btn-action" style="width:100%; background:#f1f5f9">Ouvrir</button></a>
<button class="btn-action delete-btn" onclick="removeResource(${resource.id})">Suppr.</button>
</div>
`;
resourceGrid.appendChild(card);
}

// --- COPIE PRESSE-PAPIER ---
function copyLink(link, btn) {
navigator.clipboard.writeText(link).then(() => {
const originalText = btn.innerText;
btn.innerText = "Copié ! ✅";
btn.style.background = "var(--success)";
btn.style.color = "white";

setTimeout(() => {
btn.innerText = originalText;
btn.style.background = "";
btn.style.color = "";
}, 2000);
});
}

// --- LOCAL STORAGE ---
function saveLocalResource(res) {
let list = JSON.parse(localStorage.getItem('resources') || '[]');
list.push(res);
localStorage.setItem('resources', JSON.stringify(list));
}

function getResources() {
JSON.parse(localStorage.getItem('resources') || '[]').forEach(createCard);
}

function removeResource(id) {
document.querySelector(`[data-id="${id}"]`).remove();
let list = JSON.parse(localStorage.getItem('resources'));
localStorage.setItem('resources', JSON.stringify(list.filter(r => r.id !== id)));
}
